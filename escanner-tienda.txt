<div id="lista"></div>

<button onclick="exportarCSV()">Exportar CSV</button>

<script>
let datos = JSON.parse(localStorage.getItem("tickets")) || [];

function guardar() {
  const fecha = document.getElementById("fecha").value;
  const proveedor = document.getElementById("proveedor").value;
  const categoria = document.getElementById("categoria").value;
  const monto = document.getElementById("monto").value;
  const fotoInput = document.getElementById("foto");

  const reader = new FileReader();
  reader.onload = function(e) {
    const nuevo = {
      fecha,
      proveedor,
      categoria,
      monto,
      foto: e.target.result
    };
    datos.push(nuevo);
    localStorage.setItem("tickets", JSON.stringify(datos));
    mostrar();
  };

  if (fotoInput.files[0]) {
    reader.readAsDataURL(fotoInput.files[0]);
  } else {
    alert("Sube una foto del ticket");
  }
}

function mostrar() {
  const lista = document.getElementById("lista");
  lista.innerHTML = "";
  datos.forEach(d => {
    lista.innerHTML += `
      <div class="card">
        <strong>${d.proveedor}</strong><br>
        Fecha: ${d.fecha}<br>
        Categoría: ${d.categoria}<br>
        Monto: €${d.monto}<br>
        <img src="${d.foto}">
      </div>
    `;
  });
}

function exportarCSV() {
  let csv = "Fecha,Proveedor,Categoria,Monto\n";
  datos.forEach(d => {
    csv += `${d.fecha},${d.proveedor},${d.categoria},${d.monto}\n`;
  });

  const blob = new Blob([csv], { type: "text/csv" });
  const url = window.URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = "facturas.csv";
  a.click();
}

mostrar();
</script>
